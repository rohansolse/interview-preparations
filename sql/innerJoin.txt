SELECT 
    pp.participant_id,
    pp.full_name,
    pmi.upid,
    pmi.program_type,
    pmi.oti_card
FROM program_participants pp
LEFT JOIN participants_migration_info pmi
    ON pp.application_id = pmi.application_id
WHERE pmi.program_type = 'DAILY';
